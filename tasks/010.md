### Задача 5: Публикация поста в Twitter с обработкой ошибок

### Решение задачи

```javascript
// Задача 5: Публикация поста с async/await и обработкой ошибок try...catch

// 1. Асинхронная функция, имитирующая отправку твита
async function postTweet(tweetText) {
  console.log(`Отправка твита: "${tweetText}"...`);
  // 2. Возвращаем Promise, который может разрешиться или отклониться
  return new Promise((resolve, reject) => {
    // Имитация сетевой задержки
    setTimeout(() => {
      const successChance = Math.random(); // Случайное число от 0 до 1
      if (successChance <= 0.7) { // 70% шанс на успех
        resolve(`Твит '${tweetText}' успешно опубликован!`);
      } else { // 30% шанс на ошибку
        reject(new Error("Ошибка сети при публикации"));
      }
    }, 1000); // Задержка 1 секунда
  });
}

// 3. Основная асинхронная функция для запуска процесса
async function publish() {
  try {
    // 4. Пытаемся опубликовать твит и ждем результат
    const result = await postTweet("Мой первый твит с async/await!");
    // 5. Если await не выбросил ошибку, значит промис разрешился
    console.log("Успех:", result);
  } catch (error) {
    // 6. Если await выбросил ошибку (промис отклонился), ловим ее здесь
    console.error("Не удалось опубликовать:", error.message);
  } finally {
    // 7. Этот блок выполнится в любом случае (после try или catch)
    console.log("Операция публикации завершена (успех или ошибка).");
  }
}

// 8. Запускаем процесс публикации
console.log("Попытка публикации твита...");
publish();

// Возможный вывод при УСПЕХЕ:
// Попытка публикации твита...
// Отправка твита: "Мой первый твит с async/await!"...
// (через 1 секунду)
// Успех: Твит 'Мой первый твит с async/await!' успешно опубликован!
// Операция публикации завершена (успех или ошибка).

// Возможный вывод при ОШИБКЕ:
// Попытка публикации твита...
// Отправка твита: "Мой первый твит с async/await!"...
// (через 1 секунду)
// Не удалось опубликовать: Ошибка сети при публикации
// Операция публикации завершена (успех или ошибка).

```
