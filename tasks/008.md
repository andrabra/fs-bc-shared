### Задача 3: Загрузка данных для чата в Telegram

### Решение задачи

```javascript
// Задача 3: Параллельная загрузка данных с Promise.all()

// 1. Функции, имитирующие асинхронную загрузку данных
function fetchMessages() {
  return new Promise(resolve => {
    // Имитация сетевой задержки в 1 секунду
    setTimeout(() => {
      console.log("Ответ от сервера: Сообщения получены");
      resolve("Сообщения загружены");
    }, 1000);
  });
}

function fetchStatuses() {
  return new Promise(resolve => {
    // Имитация сетевой задержки в 500 мс
    setTimeout(() => {
      console.log("Ответ от сервера: Статусы получены");
      resolve("Статусы загружены");
    }, 500);
  });
}

function fetchStickers() {
  return new Promise(resolve => {
    // Имитация сетевой задержки в 1.5 секунды
    setTimeout(() => {
      console.log("Ответ от сервера: Стикеры получены");
      resolve("Стикеры загружены");
    }, 1500);
  });
}

// 4. Начало процесса
console.log("Начало загрузки данных чата");

// Массив промисов
const promises = [
  fetchMessages(),
  fetchStatuses(),
  fetchStickers()
];

// 4. Сообщение о том, что запросы инициированы
console.log("Запросы на загрузку отправлены (Promise.all инициирован)");

// 2. Используем Promise.all для параллельного выполнения
Promise.all(promises)
  .then(results => {
    // 3. Этот код выполнится, когда ВСЕ промисы успешно завершатся
    console.log("Все данные успешно загружены:");
    console.log(results); // Выводим массив результатов
    console.log("Чат готов к отображению!");
  })
  .catch(error => {
    // Обработка возможной ошибки (хотя в этом примере ее нет)
    console.error("Ошибка при загрузке данных чата:", error);
  });

// Ожидаемый порядок вывода (примерный, зависит от точных таймингов):
// 1. Начало загрузки данных чата
// 2. Запросы на загрузку отправлены (Promise.all инициирован)
// 3. (через ~500мс) Ответ от сервера: Статусы получены
// 4. (через ~1000мс) Ответ от сервера: Сообщения получены
// 5. (через ~1500мс) Ответ от сервера: Стикеры получены
// 6. (сразу после 5) Все данные успешно загружены:
// 7. ["Сообщения загружены", "Статусы загружены", "Стикеры загружены"]
// 8. Чат готов к отображению!

```
